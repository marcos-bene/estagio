    -- Tabela Endereco
    CREATE TABLE Endereco (
        UniqueID INT AUTO_INCREMENT,
        rua VARCHAR(50) NOT NULL,
        numero VARCHAR(20) NOT NULL,
        estado CHAR(2) NOT NULL, -- O estado geralmente é representado por uma sigla de 2 caracteres
        cidade VARCHAR(50) NOT NULL,
        PRIMARY KEY (UniqueID)
    );

    -- Tabela Cliente
    CREATE TABLE Cliente (
        UniqueID INT AUTO_INCREMENT,
        nome VARCHAR(60) NOT NULL,
        sexo CHAR(1) NOT NULL, -- O sexo geralmente é representado por um único caractere, 'M' ou 'F'
        dataNascimento DATE NOT NULL,
        cpf CHAR(11) NOT NULL,
        endereco INT NOT NULL,
        telefone VARCHAR(14) NOT NULL, -- O telefone deve ser VARCHAR para acomodar diferentes formatos
        PRIMARY KEY (UniqueID),
        FOREIGN KEY (endereco) REFERENCES Endereco(UniqueID)
    );

    -- Tabela Usuarios
    CREATE TABLE Usuarios (
        UserID INT AUTO_INCREMENT PRIMARY KEY,
        Username VARCHAR(50) UNIQUE NOT NULL,
        Email VARCHAR(100) UNIQUE NOT NULL,
        PasswordHash CHAR(64) NOT NULL
    );


    -- Tabela Fornecedor
    CREATE TABLE Fornecedor (
        UniqueID INT AUTO_INCREMENT,
        nome VARCHAR(60) NOT NULL,
        sexo CHAR(1) NOT NULL,
        dataNascimento DATE NOT NULL,
        cpf CHAR(11) NOT NULL,
        endereco INT NOT NULL,
        telefone VARCHAR(14) NOT NULL,
        PRIMARY KEY (UniqueID),
        FOREIGN KEY (endereco) REFERENCES Endereco(UniqueID)
    );

        -- Tabela Produto
        CREATE TABLE Produto (
            UniqueID INT AUTO_INCREMENT,
            nome VARCHAR(50) NOT NULL,
            fornecedor INT NOT NULL,
            descricao VARCHAR(100) NOT NULL,
            categoria CHAR(10) NOT NULL,
            dataCompra DATE NOT NULL,
            qtdProduto INT NOT NULL,
            preco DECIMAL(10,2) NOT NULL, -- Usar DECIMAL para valores monetários
            PRIMARY KEY (UniqueID),
            FOREIGN KEY (fornecedor) REFERENCES Fornecedor(UniqueID)
        );
